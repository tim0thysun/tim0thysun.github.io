      A1: Blink!
      Timothy Sun

      Adapted from code tutorial (http://www.arduino.cc/en/Tutorial/Blink) by 
      Scott Fitzgerald, Arturo Guadalupi, and Colby Newman

      Turns on 4 LEDs one by one in sequence from left to right,
      then turns on all 4 LEDs at the same time.
      Turns on 4 LEDs one by one in sequence from right to left,
      then turns on all 4 LEDs at the same time.
      
    */

    // the setup function runs once when you press reset or power the board
    void setup() {
      // initialize digital pin LED_BUILTIN as an output.
      pinMode(LED_BUILTIN, OUTPUT);
      // initialize digital pin 9 as an output.
      pinMode(9, OUTPUT);
      // initialize digital pin 10 as an output.
      pinMode(10, OUTPUT);
      // initialize digital pin 11 as an output.
      pinMode(11, OUTPUT);
    }

    // the loop function runs over and over again forever
    void loop() {

      // turns on LEDS from left to right, one by one (next 8 lines)

      // turn the LED (on pin 13) on by making voltage HIGH
      digitalWrite(LED_BUILTIN, HIGH);
      // wait for 0.3 seconds
      delay(300);
      // turn the LED (on pin 11) on by making voltage HIGH  
      digitalWrite(11, HIGH); 
      // wait for 0.3 seconds
      delay(300);
      // turn the LED (on pin 10) on by making voltage HIGH 
      digitalWrite(10 , HIGH);
      // wait for 0.3 seconds
      delay(300);
      // turn the LED (on pin 9) on by making voltage HIGH     
      digitalWrite(9, HIGH);
      // wait for 0.3 seconds
      delay(300);    

      // turns off all LEDs at once (next 5 lines)

      // turn the LED (on pin 13) off by making voltage LOW
      digitalWrite(LED_BUILTIN, LOW);
      // turn the LED (on pin 11) on by making voltage LOW 
      digitalWrite(11, LOW); 
      // turn the LED (on pin 10) on by making voltage LOW
      digitalWrite(10 , LOW);
      // turn the LED (on pin 9) on by making voltage LOW
      digitalWrite(9, LOW);
      // wait for 0.3 seconds  
      delay(300);

      // turns on all LEDs at once (next 5 lines)

      // turn the LED (on pin 13) on by making voltage HIGH
      digitalWrite(LED_BUILTIN, HIGH);
      // turn the LED (on pin 11) on by making voltage HIGH 
      digitalWrite(11, HIGH); 
      // turn the LED (on pin 10) on by making voltage HIGH
      digitalWrite(10 , HIGH);
      // turn the LED (on pin 9) on by making voltage HIGH
      digitalWrite(9, HIGH);
      // wait for 0.3 seconds
      delay(300);  

      // turns off all LEDs at once (next 5 lines)

      // turn the LED (on pin 13) off by making voltage LOW
      digitalWrite(LED_BUILTIN, LOW); 
      // turn the LED (on pin 11) off by making voltage LOW
      digitalWrite(11, LOW); 
      // turn the LED (on pin 10) off by making voltage LOW
      digitalWrite(10 , LOW);
      // turn the LED (on pin 9) off by making voltage LOW
      digitalWrite(9, LOW);  
      // wait for 0.3 seconds
      delay(300);

      // turns on LEDS from right to left, one by one (next 8 lines)

      // turn the LED (on pin 9) on by making voltage HIGH
      digitalWrite(9, HIGH);
      // wait for 0.3 seconds
      delay(300);  
      // turn the LED (on pin 10) on by making voltage HIGH
      digitalWrite(10, HIGH); 
      // wait for 0.3 seconds
      delay(300);   
      // turn the LED (on pin 11) on by making voltage HIGH
      digitalWrite(11, HIGH);
      // wait for 0.3 seconds
      delay(300);   
      // turn the LED (on pin 13) on by making voltage HIGH   
      digitalWrite(LED_BUILTIN, HIGH);
      // wait for 0.3 seconds
      delay(300);    

      // turns off all LEDs at once (next 5 lines)
      
      // turn the LED (on pin 13) off by making voltage LOW
      digitalWrite(LED_BUILTIN, LOW); 
      // turn the LED (on pin 11) off by making voltage LOW
      digitalWrite(11, LOW); 
      // turn the LED (on pin 10) off by making voltage LOW
      digitalWrite(10 , LOW);
      // turn the LED (on pin 9) off by making voltage LOW
      digitalWrite(9, LOW);  
      // wait for 0.3 seconds
      delay(300);

      // turns on all LEDs at once (next 5 lines)

      // turn the LED (on pin 13) on by making voltage HIGH
      digitalWrite(LED_BUILTIN, HIGH);
      // turn the LED (on pin 11) on by making voltage HIGH 
      digitalWrite(11, HIGH); 
      // turn the LED (on pin 10) on by making voltage HIGH
      digitalWrite(10 , HIGH);
      // turn the LED (on pin 9) on by making voltage HIGH
      digitalWrite(9, HIGH);
      // wait for 0.3 seconds
      delay(300);

      // turns off all LEDs at once (next 5 lines)
      
      // turn the LED (on pin 13) off by making voltage LOW
      digitalWrite(LED_BUILTIN, LOW); 
      // turn the LED (on pin 11) off by making voltage LOW
      digitalWrite(11, LOW); 
      // turn the LED (on pin 10) off by making voltage LOW
      digitalWrite(10 , LOW);
      // turn the LED (on pin 9) off by making voltage LOW
      digitalWrite(9, LOW);  
      // wait for 0.3 seconds
      delay(300);

    }
